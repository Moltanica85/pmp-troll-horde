// Konfiguracja klatek - ZMIERZ SWÓJ ARKUSZ I WPISZ TUTAJ WARTOŚCI const FRAME_WIDTH = 128; // Szerokość jednej klatki (jednego ludzika) const FRAME_HEIGHT = 128; // Wysokość jednej klatki (jednego ludzika) class SelectionScene extends Phaser.Scene { constructor() { super("SelectionScene"); } preload() { // Ładowanie arkuszy z parametrami cięcia (frameConfig) // Zakładam, że pliki nazywają się rysiek.png i slawek.png this.load.spritesheet("rysiek_sheet", "assets/rysiek.png", { frameWidth: FRAME_WIDTH, frameHeight: FRAME_HEIGHT }); this.load.spritesheet("slawek_sheet", "assets/slawek.png", { frameWidth: FRAME_WIDTH, frameHeight: FRAME_HEIGHT }); } create() { this.add.text(this.scale.width / 2, 100, "WYBIERZ BOHATERA", { fontSize: "32px", fill: "#fff" }).setOrigin(0.5); // Tworzenie animacji "idle" (oddech/stanie w miejscu) this.anims.create({ key: 'rysiek_idle', frames: this.anims.generateFrameNumbers('rysiek_sheet', { start: 0, end: 3 }), // pierwsze 4 klatki frameRate: 8, repeat: -1 }); this.anims.create({ key: 'slawek_idle', frames: this.anims.generateFrameNumbers('slawek_sheet', { start: 0, end: 3 }), frameRate: 8, repeat: -1 }); // Interaktywne postacie let rysiek = this.add.sprite(this.scale.width * 0.35, this.scale.height / 2, "rysiek_sheet").setInteractive(); rysiek.play('rysiek_idle'); this.add.text(this.scale.width * 0.35, this.scale.height / 2 + 100, "RYSIEK", { fontSize: "24px" }).setOrigin(0.5); let slawek = this.add.sprite(this.scale.width * 0.65, this.scale.height / 2, "slawek_sheet").setInteractive(); slawek.play('slawek_idle'); this.add.text(this.scale.width * 0.65, this.scale.height / 2 + 100, "SŁAWEK", { fontSize: "24px" }).setOrigin(0.5); // Logika kliknięcia - przejście do gry rysiek.on('pointerdown', () => { this.scene.start("GameScene", { selectedHero: "rysiek_sheet" }); }); slawek.on('pointerdown', () => { this.scene.start("GameScene", { selectedHero: "slawek_sheet" }); }); } }
