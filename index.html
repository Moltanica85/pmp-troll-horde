let globalVolume = 0.3; const CONFIG = { type: Phaser.AUTO, scale: { mode: Phaser.Scale.FIT, autoCenter: Phaser.Scale.CENTER_BOTH, width: 1280, height: 720 }, physics: { default: 'arcade', arcade: { gravity: { y: 0 } } }, scene: null }; class BootScene extends Phaser.Scene { constructor() { super("BootScene"); } preload() { this.load.image('oknoglowne', 'assets/oknoglowne.png'); ['tlolv1', 'tlolv2', 'tlolv3'].forEach(bg => this.load.image(bg, `assets/${bg}.jpeg`)); this.load.image('hpower2', 'assets/hpower2.png'); this.load.image('opower1', 'assets/opower1.png'); // PRECYZYJNE WYMIARY KLATKI DLA TWOICH PLIKÓW const f = { frameWidth: 512, frameHeight: 768 }; this.load.spritesheet('p_idle', 'assets/bohater1panel1.png', f); this.load.spritesheet('p_blue', 'assets/bohater1panel2.png', f); this.load.spritesheet('p_green', 'assets/bohater1panel3.png', f); this.load.spritesheet('p_death', 'assets/bohater1panel4.png', f); this.load.spritesheet('p_run', 'assets/bohater1panel5.png', f); for(let i=1; i<=3; i++) { this.load.image('tb'+i, `assets/tb${i}.png`); } this.load.audio('track1', 'assets/muzyka1.mp3'); this.load.audio('track2', 'assets/muzyka2.mp3'); } create() { // Definicje animacji (klatki 0-3 to pierwszy rząd Twoich grafik) this.anims.create({ key: 'idle', frames: this.anims.generateFrameNumbers('p_idle', { start: 0, end: 3 }), frameRate: 6, repeat: -1 }); this.anims.create({ key: 'run', frames: this.anims.generateFrameNumbers('p_run', { start: 0, end: 3 }), frameRate: 10, repeat: -1 }); this.anims.create({ key: 'atk_blue', frames: this.anims.generateFrameNumbers('p_blue', { start: 0, end: 3 }), frameRate: 12 }); this.anims.create({ key: 'atk_green', frames: this.anims.generateFrameNumbers('p_green', { start: 0, end: 3 }), frameRate: 12 }); this.anims.create({ key: 'die', frames: this.anims.generateFrameNumbers('p_death', { start: 8, end: 11 }), frameRate: 8 }); this.scene.start("StartScene"); } } class StartScene extends Phaser.Scene { constructor() { super("StartScene"); } create() { this.add.image(640, 360, 'oknoglowne').setDisplaySize(1280, 720); // Przycisk START let btn = this.add.rectangle(640, 400, 300, 80, 0x000000, 0.7).setInteractive().setStrokeStyle(4, 0xffffff); this.add.text(640, 400, "START", { fontSize: '42px', fill: '#fff' }).setOrigin(0.5); btn.on('pointerdown', () => { if (!this.sound.get('track1')) this.sound.play('track1', { loop: true, volume: globalVolume }); this.scene.start("SelectScene"); }); // Suwak głośności this.add.text(640, 550, "GŁOŚNOŚĆ", { fontSize: '20px' }).setOrigin(0.5); let track = this.add.rectangle(640, 600, 200, 10, 0x333333); let handle = this.add.circle(540 + (globalVolume * 200), 600, 15, 0xffffff).setInteractive(); this.input.setDraggable(handle); this.input.on('drag', (pointer, gameObject, dragX) => { dragX = Phaser.Math.Clamp(dragX, 540, 740); gameObject.x = dragX; globalVolume = (dragX - 540) / 200; this.sound.setVolume(globalVolume); }); } } class SelectScene extends Phaser.Scene { constructor() { super("SelectScene"); } create() { this.add.text(640, 100, "WYBIERZ BOHATERA", { fontSize: '40px', fill: '#fff' }).setOrigin(0.5); // Wyświetlamy konkretnie klatkę 0 (sprite, nie cały arkusz) let r = this.add.sprite(440, 400, 'p_idle', 0).setScale(0.4).setInteractive(); this.add.text(440, 600, "RYSIEK", { fontSize: '30px' }).setOrigin(0.5); r.on('pointerdown', () => this.scene.start("GameScene", { hero: 'rysiek', tint: 0xffffff, special: 'atk_blue' })); let s = this.add.sprite(840, 400, 'p_idle', 0).setScale(0.4).setInteractive().setTint(0x99ff99); this.add.text(840, 600, "SŁAWEK", { fontSize: '30px' }).setOrigin(0.5); s.on('pointerdown', () => this.scene.start("GameScene", { hero: 'slawek', tint: 0x99ff99, special: 'atk_green' })); } } class GameScene extends Phaser.Scene { constructor() { super("GameScene"); } init(data) { this.heroData = data; this.score = 0; this.isDead = false; } create() { this.add.image(640, 360, 'tlolv1').setDisplaySize(1280, 720); // Gracz z poprawionym Body (kolizje) this.player = this.physics.add.sprite(200, 400, 'p_idle', 0).setScale(0.35); this.player.setTint(this.heroData.tint).setCollideWorldBounds(true); this.player.body.setSize(250, 450).setOffset(130, 150); this.trolls = this.physics.add.group(); this.bullets = this.physics.add.group(); this.keys = this.input.keyboard.addKeys('W,A,S,D,K,L'); this.time.addEvent({ delay: 1500, callback: this.spawnTroll, callbackScope: this, loop: true }); this.ui = this.add.text(20, 20, '', { fontSize: '28px', fill: '#fff', stroke: '#000', strokeThickness: 4 }); this.physics.add.overlap(this.bullets, this.trolls, (b, t) => { b.destroy(); t.destroy(); this.score++; }); this.physics.add.overlap(this.player, this.trolls, () => this.die()); } spawnTroll() { if(this.isDead) return; let t = this.trolls.create(1350, Phaser.Math.Between(200, 600), 'tb1').setScale(0.3).setFlipX(true); t.setVelocityX(-200 - (this.score * 5)); } update() { if(this.isDead) return; this.ui.setText(`BOHATER: ${this.heroData.hero.toUpperCase()} | PUNKTY: ${this.score}`); let vx = 0, vy = 0; if(this.keys.A.isDown) vx = -400; if(this.keys.D.isDown) vx = 400; if(this.keys.W.isDown) vy = -400; if(this.keys.S.isDown) vy = 400; this.player.setVelocity(vx, vy); if(vx !== 0 || vy !== 0) this.player.play('run', true); else this.player.play('idle', true); if(Phaser.Input.Keyboard.JustDown(this.keys.K)) this.shoot(); } shoot() { let b = this.bullets.create(this.player.x + 50, this.player.y, 'hpower2').setScale(0.2); b.setVelocityX(800); this.player.play(this.heroData.special, true); } die() { if(this.isDead) return; this.isDead = true; this.player.setVelocity(0).play('die'); this.time.delayedCall(2000, () => this.scene.start("StartScene")); } } const game = new Phaser.Game(CONFIG);
