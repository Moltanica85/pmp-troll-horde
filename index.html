let globalVolume = 0.3; const CONFIG = { type: Phaser.AUTO, scale: { mode: Phaser.Scale.FIT, autoCenter: Phaser.Scale.CENTER_BOTH, width: 1280, height: 720 }, physics: { default: 'arcade', arcade: { gravity: { y: 0 } } }, scene: null }; class BootScene extends Phaser.Scene { constructor() { super("BootScene"); } preload() { // Tła i UI this.load.image('oknoglowne', 'assets/oknoglowne.png'); this.load.image('tabelarekordow', 'assets/tabelarekordow.png'); ['tlolv1', 'tlolv2', 'tlolv3'].forEach(bg => this.load.image(bg, `assets/${bg}.jpeg`)); this.load.image('hpower2', 'assets/hpower2.png'); this.load.image('opower1', 'assets/opower1.png'); // PANELE POSTACI - Ustalamy sztywno wymiary klatki (512x768) const frameConfig = { frameWidth: 512, frameHeight: 768 }; this.load.spritesheet('p_idle', 'assets/bohater1panel1.png', frameConfig); this.load.spritesheet('p_blue', 'assets/bohater1panel2.png', frameConfig); this.load.spritesheet('p_green', 'assets/bohater 1panel3png.png', frameConfig); this.load.spritesheet('p_death', 'assets/bohater1panel4.png', frameConfig); this.load.spritesheet('p_run', 'assets/bohater1panel5.png', frameConfig); // Przeciwnicy for(let i=1; i<=3; i++) { this.load.image('tb'+i, `assets/tb${i}.png`); this.load.image('tk'+i, `assets/tk${i}.png`); this.load.image('tm'+i, `assets/tm${i}.png`); } // Dźwięk this.load.audio('track1', 'assets/muzyka1.mp3'); this.load.audio('track2', 'assets/muzyka2.mp3'); } create() { // Animacje - poprawione indeksy klatek dla Twoich arkuszy 4x4 this.anims.create({ key: 'idle', frames: this.anims.generateFrameNumbers('p_idle', { start: 0, end: 3 }), frameRate: 6, repeat: -1 }); this.anims.create({ key: 'run', frames: this.anims.generateFrameNumbers('p_run', { start: 0, end: 3 }), frameRate: 10, repeat: -1 }); this.anims.create({ key: 'atk_k', frames: this.anims.generateFrameNumbers('p_idle', { start: 8, end: 11 }), frameRate: 12 }); this.anims.create({ key: 'atk_blue', frames: this.anims.generateFrameNumbers('p_blue', { start: 4, end: 7 }), frameRate: 10 }); this.anims.create({ key: 'atk_green', frames: this.anims.generateFrameNumbers('p_green', { start: 0, end: 3 }), frameRate: 10 }); this.anims.create({ key: 'die', frames: this.anims.generateFrameNumbers('p_death', { start: 8, end: 11 }), frameRate: 8 }); this.scene.start("StartScene"); } } class StartScene extends Phaser.Scene { constructor() { super("StartScene"); } create() { this.sound.stopAll(); if (this.cache.audio.exists('track1')) { this.sound.play('track1', { loop: true, volume: globalVolume }); } this.add.image(640, 360, 'oknoglowne').setDisplaySize(1280, 720); let startBtn = this.add.rectangle(640, 480, 250, 80, 0x000000, 0.7).setInteractive().setStrokeStyle(4, 0xffffff); this.add.text(640, 480, "START", { fontSize: '42px', color: '#fff' }).setOrigin(0.5); startBtn.on('pointerdown', () => this.scene.start("SelectScene")); // Suwak głośności (prosty tekstowy na próbę) let volText = this.add.text(640, 600, `GŁOŚNOŚĆ: ${Math.round(globalVolume * 100)}%`, { fontSize: '20px' }).setOrigin(0.5); this.input.on('pointerdown', (pointer) => { if(pointer.y > 580 && pointer.y < 620) { globalVolume = Phaser.Math.Clamp((pointer.x - 540) / 200, 0, 1); this.sound.setVolume(globalVolume); volText.setText(`GŁOŚNOŚĆ: ${Math.round(globalVolume * 100)}%`); } }); } } class SelectScene extends Phaser.Scene { constructor() { super("SelectScene"); } create() { this.add.text(640, 100, "WYBIERZ BOHATERA", { fontSize: '40px', color: '#fff' }).setOrigin(0.5); // RYSIEK - Wyświetlamy tylko PIERWSZĄ klatkę, nie cały obrazek let rysiek = this.add.sprite(400, 350, 'p_idle', 0).setScale(0.5).setInteractive(); this.add.text(400, 550, "RYSIEK", { fontSize: '30px' }).setOrigin(0.5); rysiek.on('pointerdown', () => this.scene.start("GameScene", { hero: 'rysiek', skin: null, special: 'atk_blue' })); // SŁAWEK let slawek = this.add.sprite(880, 350, 'p_idle', 0).setScale(0.5).setInteractive().setTint(0x99ff99); this.add.text(880, 550, "SŁAWEK", { fontSize: '30px' }).setOrigin(0.5); slawek.on('pointerdown', () => this.scene.start("GameScene", { hero: 'slawek', skin: 0x99ff99, special: 'atk_green' })); } } class GameScene extends Phaser.Scene { constructor() { super("GameScene"); } init(data) { this.hero = data; this.score = 0; this.isDead = false; this.isAttacking = false; } create() { this.bg = this.add.image(640, 360, 'tlolv1').setDisplaySize(1280, 720); // Gracz z jedną klatką na start this.player = this.physics.add.sprite(200, 400, 'p_idle', 0).setScale(0.4); if(this.hero.skin) this.player.setTint(this.hero.skin); this.player.setCollideWorldBounds(true).setSize(200, 400); this.trolls = this.physics.add.group(); this.bullets = this.physics.add.group(); this.keys = this.input.keyboard.addKeys('W,A,S,D,K,L'); this.ui = this.add.text(20, 20, 'Trolle: 0', { fontSize: '30px', color: '#fff' }); this.time.addEvent({ delay: 1500, callback: this.spawnTroll, callbackScope: this, loop: true }); this.physics.add.overlap(this.bullets, this.trolls, (b, t) => { b.destroy(); t.destroy(); this.score++; }); this.physics.add.overlap(this.player, this.trolls, () => this.playerDie()); } spawnTroll() { if(this.isDead) return; let t = this.trolls.create(1300, Phaser.Math.Between(150, 600), 'tb1').setScale(0.3); t.setFlipX(true).setVelocityX(-250); } update() { if(this.isDead) return; this.ui.setText(`RYCERZ: ${this.hero.hero.toUpperCase()} | TROLLE: ${this.score}`); let vx = 0, vy = 0; if(this.keys.A.isDown) vx = -400; if(this.keys.D.isDown) vx = 400; if(this.keys.W.isDown) vy = -400; if(this.keys.S.isDown) vy = 400; this.player.setVelocity(vx, vy); if(!this.isAttacking) { if(vx !== 0 || vy !== 0) this.player.play('run', true); else this.player.play('idle', true); } if(Phaser.Input.Keyboard.JustDown(this.keys.K)) this.attackK(); if(Phaser.Input.Keyboard.JustDown(this.keys.L)) this.attackL(); } attackK() { if(this.isAttacking) return; this.isAttacking = true; this.player.play('atk_k'); let b = this.bullets.create(this.player.x + 50, this.player.y, 'hpower2').setScale(0.2); b.setVelocityX(800); this.time.delayedCall(500, () => this.isAttacking = false); } attackL() { if(this.isAttacking) return; this.isAttacking = true; this.player.play(this.hero.special); // Efekt obszarowy let boom = this.add.image(this.player.x, this.player.y, 'opower1').setScale(0.1); this.tweens.add({ targets: boom, scale: 4, alpha: 0, duration: 500, onComplete: () => boom.destroy() }); this.trolls.children.iterate(t => { if(t && Phaser.Math.Distance.Between(this.player.x, this.player.y, t.x, t.y) < 250) t.destroy(); }); this.time.delayedCall(800, () => this.isAttacking = false); } playerDie() { if(this.isDead) return; this.isDead = true; this.player.setVelocity(0).play('die'); this.time.delayedCall(2000, () => this.scene.start("StartScene")); } } const game = new Phaser.Game(Object.assign(CONFIG, { scene: [BootScene, StartScene, SelectScene, GameScene] }));
