// --- INTELIGENTNY SPAWN TROLLA --- spawnTroll(types) { const type = Phaser.Utils.Array.GetRandom(types); const variant = Phaser.Math.Between(1, 3); let key = (type === 'bagienny') ? `tb${variant}` : (type === 'kamienny' ? `tk${variant}` : `tm${variant}`); let troll = this.trolls.create(this.scale.width + 50, Phaser.Math.Between(100, this.scale.height - 100), key); // Nadawanie właściwości fizycznych troll.setFlipX(true).setScale(0.35); troll.setCircle(120); // Lepsza kolizja niż kwadrat troll.hp = (type === 'kamienny') ? 3 : 1; // Kamienne są twardsze // Ruch falowy (Sine Wave) - Zamiast nudnego lotu prosto this.tweens.add({ targets: troll, y: '+=50', duration: 1500, yoyo: true, repeat: -1, ease: 'Sine.easeInOut' }); troll.body.setVelocityX(type === 'kamienny' ? -100 : -180); } // --- POPRAWIONY STRZAŁ Z EFEKTAMI --- useImpulse() { if (this.player.cooldowns.impulse > Date.now()) return; this.player.cooldowns.impulse = Date.now() + 500; let proj = this.projectiles.create(this.player.x + 40, this.player.y, 'hpower2'); proj.setScale(0.15).setTint(0x00ffff); proj.body.setVelocityX(800); // Dodanie lekkiego odrzutu dla gracza (feeling!)[cite: 3] this.player.x -= 5; // Kolizja z trollami[cite: 3] this.physics.add.overlap(proj, this.trolls, (p, t) => { p.destroy(); t.hp -= 1; t.setTint(0xffffff); // Mignięcie przy trafieniu this.time.delayedCall(100, () => t.clearTint()); // Knockback - odrzucenie trolla do tyłu[cite: 3] t.x += 20; if (t.hp <= 0) { // Efekt cząsteczek przy śmierci[cite: 3] this.cameras.main.shake(100, 0.005); t.destroy(); } }); }
