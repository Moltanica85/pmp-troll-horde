class BootScene extends Phaser.Scene { constructor() { super('BootScene'); } preload() { // Tła dla wszystkich poziomów this.load.image('oknoglowne', 'assets/oknoglowne.png'); this.load.image('tlolv1', 'assets/tlolv1.jpeg'); this.load.image('tlolv2', 'assets/tlolv2.jpeg'); this.load.image('tlolv3', 'assets/tlolv3.jpeg'); // Obiekty i ikony this.load.image('hpower2', 'assets/hpower2.png'); this.load.image('opower1', 'assets/opower1.png'); // BOHATER GŁÓWNY - Ładowanie jako spritesheet (rozmiar klatki 200x200px) this.load.spritesheet('hero_sheet', 'assets/bh1run.png', { frameWidth: 200, frameHeight: 200 }); // TROLE - Poziom 1 (tb), Poziom 2 (tk), Poziom 3 (tm) this.load.image('tb1', 'assets/tb1.png'); this.load.image('tb2', 'assets/tb2.png'); this.load.image('tb3', 'assets/tb3.png'); this.load.image('tk1', 'assets/tk1.png'); this.load.image('tk2', 'assets/tk2.png'); this.load.image('tk3', 'assets/tk3.png'); this.load.image('tm1', 'assets/tm1.png'); this.load.image('tm2', 'assets/tm2.png'); this.load.image('tm3', 'assets/tm3.png'); // Pliki audio this.load.audio('track1', 'assets/muzyka1.mp3'); this.load.audio('track2', 'assets/muzyka2.mp3'); } create() { // Definicje animacji dla bohatera this.anims.create({ key: 'run', frames: this.anims.generateFrameNumbers('hero_sheet', { start: 0, end: 11 }), frameRate: 12, repeat: -1 }); this.anims.create({ key: 'idle', frames: this.anims.generateFrameNumbers('hero_sheet', { start: 12, end: 15 }), frameRate: 4, repeat: -1 }); this.anims.create({ key: 'atk', frames: this.anims.generateFrameNumbers('hero_sheet', { start: 16, end: 20 }), frameRate: 14, repeat: 0 }); // Animacja specjalna: "moc bieli" (klatki od 21 do 24) this.anims.create({ key: 'hero-power', frames: this.anims.generateFrameNumbers('hero_sheet', { start: 21, end: 24 }), frameRate: 10, repeat: 0 }); // Przejście do głównego menu this.scene.start("StartScene"); } }
