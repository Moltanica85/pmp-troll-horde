/* =========================================================== PRZERÓB MNIE PANIE: TROLL HORDE (WERSJA GITHUB) =========================================================== */ const CONFIG = { type: Phaser.AUTO, width: window.innerWidth, height: window.innerHeight, render: { antialias: true, pixelArt: false }, physics: { default: 'arcade', arcade: { gravity: { y: 0 }, debug: false } }, scene: null }; const LEVELS = [ { name: "Ścieżka Białego Wejścia", horde: 15, types: ["bagienny"], bg: 'tlolv1' }, { name: "Kamienna Przełęcz", horde: 25, types: ["bagienny", "kamienny"], bg: 'tlolv2' }, { name: "Mchowy Amfiteatr", horde: 40, types: ["bagienny", "mchowy"], bg: 'tlolv3' } ]; // --- MENU GŁÓWNE --- class MenuScene extends Phaser.Scene { constructor() { super("MenuScene"); } preload() { // Poprawne ścieżki do Twoich plików na GitHubie this.load.image('oknoglowne', './assets/oknoglowne.png'); this.load.image('bohater1', './assets/bohater1.png'); this.load.image('bohater2', './assets/bohater2.png'); } create() { this.add.image(this.scale.width / 2, this.scale.height / 2, 'oknoglowne').setDisplaySize(this.scale.width, this.scale.height); this.add.text(this.scale.width / 2, 80, "PRZERÓB MNIE PANIE", { fontSize: '48px', fill: '#fff', fontStyle: 'bold' }).setOrigin(0.5); this.add.text(this.scale.width / 2, 140, "WYBIERZ POSTAĆ", { fontSize: '24px', fill: '#fff' }).setOrigin(0.5); let p1 = this.add.image(this.scale.width * 0.35, this.scale.height * 0.6, 'bohater1').setInteractive().setScale(0.5); let p2 = this.add.image(this.scale.width * 0.65, this.scale.height * 0.6, 'bohater2').setInteractive().setScale(0.5); p1.on('pointerdown', () => this.scene.start("GameScene", { playerKey: 'bohater1', levelIndex: 0 })); p2.on('pointerdown', () => this.scene.start("GameScene", { playerKey: 'bohater2', levelIndex: 0 })); } } // --- GRA --- class GameScene extends Phaser.Scene { constructor() { super("GameScene"); } init(data) { this.levelIndex = data.levelIndex || 0; this.playerKey = data.playerKey || 'bohater1'; this.isGameOver = false; } preload() { // Tła this.load.image('tlolv1', './assets/tlolv1.jpeg'); this.load.image('tlolv2', './assets/tlolv2.jpeg'); this.load.image('tlolv3', './assets/tlolv3.jpeg'); this.load.image('wTitle', './assets/wTitle.png'); // Moce[cite: 1] this.load.image('hpower2', './assets/hpower2.jpeg'); this.load.image('opower1', './assets/opower1.jpeg'); // Trolle - Bagienne[cite: 1] this.load.image('tb1', './assets/tb1.png'); this.load.image('tb2', './assets/tb2.png'); this.load.image('tb3', './assets/tb3.png'); // Trolle - Kamienne[cite: 1] this.load.image('tk1', './assets/tk1.jpeg'); this.load.image('tk2', './assets/tk2.jpeg'); this.load.image('tk3', './assets/tk3.jpeg'); // Trolle - Mchowe[cite: 1] this.load.image('tm1', './assets/tm1.png'); this.load.image('tm2', './assets/tm2.png'); this.load.image('tm3', './assets/tm3.png'); } create() { const lvl = LEVELS[this.levelIndex]; this.add.image(this.scale.width / 2, this.scale.height / 2, lvl.bg).setDisplaySize(this.scale.width, this.scale.height); this.player = this.physics.add.sprite(200, this.scale.height / 2, this.playerKey); this.player.setScale(0.5).setCollideWorldBounds(true); this.player.cooldowns = { flash: 0, impulse: 0 }; this.trolls = this.physics.add.group(); this.spawnTimer = this.time.addEvent({ delay: 1500, callback: () => this.spawnTroll(lvl.types), repeat: lvl.horde - 1 }); this.input.on('pointerdown', (p) => { if (p.x < this.scale.width / 2) this.useFlash(); else this.useImpulse(); }); this.physics.add.overlap(this.player, this.trolls, this.handleGameOver, null, this); } spawnTroll(types) { if (this.isGameOver) return; const type = Phaser.Utils.Array.GetRandom(types); const variant = Phaser.Math.Between(1, 3); let key = (type === 'bagienny') ? `tb${variant}` : (type === 'kamienny' ? `tk${variant}` : `tm${variant}`); let troll = this.trolls.create(this.scale.width + 100, Phaser.Math.Between(100, this.scale.height - 100), key); troll.type = type; troll.setScale(0.4); troll.body.setVelocityX(type === 'kamienny' ? -60 : -130); } useFlash() { if (this.player.cooldowns.flash > Date.now()) return; this.player.cooldowns.flash = Date.now() + 3000; let fx = this.add.image(this.player.x, this.player.y, 'opower1').setScale(0.2).setAlpha(0.8); this.tweens.add({ targets: fx, scale: 2, alpha: 0, duration: 500, onComplete: () => fx.destroy() }); this.trolls.children.iterate(t => { if(t) t.setTint(0xff0000); this.time.delayedCall(1000, () => t && t.clearTint()); }); } useImpulse() { if (this.player.cooldowns.impulse > Date.now()) return; this.player.cooldowns.impulse = Date.now() + 2000; let fx = this.add.image(this.player.x + 50, this.player.y, 'hpower2').setScale(0.5); this.tweens.add({ targets: fx, x: '+=300', alpha: 0, duration: 400, onComplete: () => fx.destroy() }); this.trolls.children.iterate(t => { if(t && Phaser.Math.Distance.Between(this.player.x, this.player.y, t.x, t.y) < 300) t.x += 150; }); } handleGameOver() { this.isGameOver = true; this.physics.pause(); this.scene.start("GameOverScene", { levelIndex: this.levelIndex, playerKey: this.playerKey }); } update() { if (this.isGameOver) return; this.trolls.children.iterate(t => { if (t && t.x < -100) this.handleGameOver(); }); if (this.spawnTimer.getRepeatCount() === 0 && this.trolls.countActive() === 0) { if (this.levelIndex < LEVELS.length - 1) this.scene.start("TransitionScene", { levelIndex: this.levelIndex + 1, playerKey: this.playerKey }); else this.scene.start("WinScene"); } } } // --- SCENY DODATKOWE --- class TransitionScene extends Phaser.Scene { constructor() { super("TransitionScene"); } create(data) { this.add.image(this.scale.width / 2, this.scale.height / 2, 'wTitle').setDisplaySize(this.scale.width, this.scale.height); this.time.delayedCall(2000, () => this.scene.start("GameScene", data)); } } class GameOverScene extends Phaser.Scene { constructor() { super("GameOverScene"); } create(data) { this.add.text(this.scale.width/2, this.scale.height/2, "PRZERÓBKA NIEUDANA", { fontSize: '40px', fill: '#f00' }).setOrigin(0.5); this.input.once('pointerdown', () => this.scene.start("GameScene", data)); } } class WinScene extends Phaser.Scene { constructor() { super("WinScene"); } create() { this.add.text(this.scale.width/2, this.scale.height/2, "KLUCZ BIELI URATOWANY!", { fontSize: '40px', fill: '#0f0' }).setOrigin(0.5); this.input.once('pointerdown', () => this.scene.start("MenuScene")); } } CONFIG.scene = [MenuScene, GameScene, TransitionScene, GameOverScene, WinScene]; new Phaser.Game(CONFIG);
