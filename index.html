let globalVolume = 0.3; const CONFIG = { type: Phaser.AUTO, scale: { mode: Phaser.Scale.FIT, autoCenter: Phaser.Scale.CENTER_BOTH, width: 1280, height: 720 }, physics: { default: 'arcade', arcade: { gravity: { y: 0 } } }, scene: null }; class BootScene extends Phaser.Scene { constructor() { super("BootScene"); } preload() { this.load.image('oknoglowne', 'assets/oknoglowne.png'); ['tlolv1', 'tlolv2', 'tlolv3'].forEach(bg => this.load.image(bg, `assets/${bg}.jpeg`)); this.load.image('hpower2', 'assets/hpower2.png'); this.load.image('opower1', 'assets/opower1.png'); // ŁADOWANIE WSZYSTKICH JAKO PNG // UWAGA: frameWidth musi być szerokością CAŁEGO obrazka podzieloną przez 4 const f = { frameWidth: 512, frameHeight: 768 }; this.load.spritesheet('p_idle', 'assets/bohater1panel1.png', f); this.load.spritesheet('p_blue', 'assets/bohater1panel2.png', f); this.load.spritesheet('p_green', 'assets/bohater1panel3.png', f); this.load.spritesheet('p_death', 'assets/bohater1panel4.png', f); this.load.spritesheet('p_run', 'assets/bohater1panel5.png', f); for(let i=1; i<=3; i++) { this.load.image('tb'+i, `assets/tb${i}.png`); } this.load.audio('track1', 'assets/muzyka1.mp3'); this.load.audio('track2', 'assets/muzyka2.mp3'); } create() { // Animacje dla siatki 4x4 this.anims.create({ key: 'idle', frames: this.anims.generateFrameNumbers('p_idle', { start: 0, end: 3 }), frameRate: 6, repeat: -1 }); this.anims.create({ key: 'run', frames: this.anims.generateFrameNumbers('p_run', { start: 0, end: 3 }), frameRate: 12, repeat: -1 }); this.anims.create({ key: 'atk_k', frames: this.anims.generateFrameNumbers('p_idle', { start: 8, end: 11 }), frameRate: 15 }); this.anims.create({ key: 'atk_blue', frames: this.anims.generateFrameNumbers('p_blue', { start: 0, end: 3 }), frameRate: 12 }); this.anims.create({ key: 'atk_green', frames: this.anims.generateFrameNumbers('p_green', { start: 0, end: 3 }), frameRate: 12 }); this.anims.create({ key: 'die', frames: this.anims.generateFrameNumbers('p_death', { start: 8, end: 11 }), frameRate: 8 }); this.scene.start("StartScene"); } } class StartScene extends Phaser.Scene { constructor() { super("StartScene"); } create() { this.sound.stopAll(); if (this.cache.audio.exists('track1')) this.sound.play('track1', { loop: true, volume: globalVolume }); this.add.image(640, 360, 'oknoglowne').setDisplaySize(1280, 720); let startBtn = this.add.rectangle(640, 480, 250, 80, 0x000000, 0.7).setInteractive().setStrokeStyle(4, 0xffffff); this.add.text(640, 480, "START", { fontSize: '42px' }).setOrigin(0.5); startBtn.on('pointerdown', () => this.scene.start("SelectScene")); // Suwak głośności let volTxt = this.add.text(640, 600, `GŁOŚNOŚĆ: ${Math.round(globalVolume * 100)}%`).setOrigin(0.5); this.input.on('pointermove', (p) => { if(p.isDown && p.y > 580) { globalVolume = Phaser.Math.Clamp((p.x - 540) / 200, 0, 1); this.sound.setVolume(globalVolume); volTxt.setText(`GŁOŚNOŚĆ: ${Math.round(globalVolume * 100)}%`); } }); } } class SelectScene extends Phaser.Scene { constructor() { super("SelectScene"); } create() { this.add.text(640, 100, "WYBIERZ BOHATERA", { fontSize: '40px' }).setOrigin(0.5); // Wyświetlamy TYLKO klatkę 0 let r = this.add.sprite(440, 350, 'p_idle', 0).setScale(0.5).setInteractive(); this.add.text(440, 550, "RYSIEK").setOrigin(0.5); r.on('pointerdown', () => this.scene.start("GameScene", { hero: 'rysiek', tint: 0xffffff, special: 'atk_blue' })); let s = this.add.sprite(840, 350, 'p_idle', 0).setScale(0.5).setInteractive().setTint(0x99ff99); this.add.text(840, 550, "SŁAWEK").setOrigin(0.5); s.on('pointerdown', () => this.scene.start("GameScene", { hero: 'slawek', tint: 0x99ff99, special: 'atk_green' })); } } class GameScene extends Phaser.Scene { constructor() { super("GameScene"); } init(data) { this.hero = data; this.score = 0; this.isDead = false; this.isAttacking = false; } create() { this.bg = this.add.image(640, 360, 'tlolv1').setDisplaySize(1280, 720); this.player = this.physics.add.sprite(200, 400, 'p_idle', 0).setScale(0.4); this.player.setTint(this.hero.tint).setCollideWorldBounds(true).setSize(200, 400); this.trolls = this.physics.add.group(); this.bullets = this.physics.add.group(); this.keys = this.input.keyboard.addKeys('W,A,S,D,K,L'); this.time.addEvent({ delay: 1500, callback: this.spawn, callbackScope: this, loop: true }); this.ui = this.add.text(20, 20, '', { fontSize: '30px', stroke: '#000', strokeThickness: 5 }); this.physics.add.overlap(this.bullets, this.trolls, (b, t) => { b.destroy(); t.destroy(); this.score++; }); this.physics.add.overlap(this.player, this.trolls, () => this.die()); } spawn() { if(this.isDead) return; let t = this.trolls.create(1300, Phaser.Math.Between(150, 600), 'tb1').setScale(0.3).setFlipX(true); t.setVelocityX(-250); } update() { if(this.isDead) return; this.ui.setText(`RYCERZ: ${this.hero.hero.toUpperCase()} | WYNIK: ${this.score}`); let vx = 0, vy = 0; if(this.keys.A.isDown) vx = -500; if(this.keys.D.isDown) vx = 500; if(this.keys.W.isDown) vy = -500; if(this.keys.S.isDown) vy = 500; this.player.setVelocity(vx, vy); if(!this.isAttacking) { if(vx !== 0 || vy !== 0) this.player.play('run', true); else this.player.play('idle', true); } if(Phaser.Input.Keyboard.JustDown(this.keys.K)) this.atkK(); if(Phaser.Input.Keyboard.JustDown(this.keys.L)) this.atkL(); } atkK() { this.isAttacking = true; this.player.play('atk_k'); let b = this.bullets.create(this.player.x + 50, this.player.y, 'hpower2').setScale(0.2); b.setVelocityX(900); this.time.delayedCall(400, () => this.isAttacking = false); } atkL() { this.isAttacking = true; this.player.play(this.hero.special); let boom = this.add.image(this.player.x, this.player.y, 'opower1').setScale(0.1); this.tweens.add({ targets: boom, scale: 5, alpha: 0, duration: 500, onComplete: () => boom.destroy() }); this.trolls.children.iterate(t => { if(t && Phaser.Math.Distance.Between(this.player.x, this.player.y, t.x, t.y) < 300) t.destroy(); }); this.time.delayedCall(600, () => this.isAttacking = false); } die() { if(this.isDead) return; this.isDead = true; this.player.setVelocity(0).play('die'); this.time.delayedCall(2000, () => this.scene.start("StartScene")); } } const game = new Phaser.Game(Object.assign(CONFIG, { scene: [BootScene, StartScene, SelectScene, GameScene] }));
